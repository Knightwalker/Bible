#include <iostream>;

class Person
{
    // Access specifier 
    public:

    // Data Members 
    std::string name;

    // Member Functions() 
    void sayHello()
    {
        std::cout << "Hello, my name is " << name << std::endl;
    }
};

int main() {
    // Declare an object of class Person 
    Person person;

    // Accessing data member 
    person.name = "Pesho";

    // Accessing member function 
    person.sayHello();

    return 0;
}
