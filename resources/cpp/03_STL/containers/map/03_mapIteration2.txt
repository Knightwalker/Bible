#include <iostream>
#include <map>
#include <string>

int main()
{
    std::map<std::string, int> gamesPriceMap {
        {"The Witcher", 100},
        {"Cyberpunk", 69},
        {"Assasins Creed", 80}
    };

    for (std::pair<std::string, int> element : gamesPriceMap) {
        std::cout << element.first << " " << element.second << std::endl;
    }

    // Assasins Creed 80
    // Cyberpunk 69
    // The Witcher 100

}
